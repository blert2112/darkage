#DarkAge MOD for MINETEST
Original mod by Master Gollum 
Modified by addi and CraigyDavi
Modified again by blert2112



###Configuration
Open init.lua
* ENABLE_MAPGEN		= true
	enable/disable map generation if you dont want to use it
* ENABLE_DECO_NODES	= true
	enable/disable decoration nodes (funiture and stuff)
* ENABLE_STAIRS		= true
	enable/disable stair registration
* ENABLE_ALIASES	= false
	enable/disable aliases
	disabled by default
